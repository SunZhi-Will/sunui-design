## Usage

```tsx
import { Card } from '@sunui-design/card';

export default function MyComponent() {
  return (
    <Card>
      <Card.Header>Card Title</Card.Header>
      <Card.Body>Card content goes here</Card.Body>
      <Card.Footer>Card footer</Card.Footer>
    </Card>
  );
}
```
